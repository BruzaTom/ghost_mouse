


def play_back(actions, window_title, repetitions):
    hwnd = win32gui.FindWindow(None, window_title)
    if hwnd:
        print(f"Window found: {hwnd}")
        for _ in range(repetitions):
            for action, pos in actions:
                # Convert (x, y) tuple to LPARAM
                lParam = win32api.MAKELONG(pos[0], pos[1])
                if action == 'move':
                    print('move')
                    win32gui.SendMessage(hwnd, win32con.WM_MOUSEMOVE, 0, lParam)
                elif action == 'click':
                    print('click')
                    win32gui.SendMessage(hwnd, win32con.WM_LBUTTONDOWN, 0, lParam)
                    time.sleep(0.1)  # Simulate a click duration
                    win32gui.SendMessage(hwnd, win32con.WM_LBUTTONUP, 0, lParam)
                elif action == 'release':
                    pass  # Handle release if needed
                time.sleep(0.1)  # Add a small delay between actions
